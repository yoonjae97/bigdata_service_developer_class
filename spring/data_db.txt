/* 새 테이블 */
CREATE TABLE data (
	no NUMBER NOT NULL, /* 일련번호 */
	subject VARCHAR2(200) NOT NULL, /* 제목 */
	content CLOB, /* 글내용 */
	userid VARCHAR2(30) NOT NULL, /* 글쓴이 */
	hit NUMBER DEFAULT 0, /* 조회수 */
	writedate DATE DEFAULT sysdate /* 등록일 */
);

ALTER TABLE data
	ADD
		CONSTRAINT PK_data
		PRIMARY KEY (
			no
		);
